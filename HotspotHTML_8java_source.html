321 String strContenuHTML = this.getStrTexteHTML();
322 File fileIndexHTML = new File(strPageHTML + File.separator + "index.html");

324 ImageEditeurHTML img1 = this.getImagesEditeur()[j];
325 strContenuHTML = strContenuHTML.replace("file:////" + img1.getStrImagePath(), "images/" + img1.getStrNomImage());

332 strContenuHTML = strContenuHTML.replace(">", ">\n");
333 fileIndexHTML.setWritable(true);
334 oswFichierHTML = new OutputStreamWriter(new FileOutputStream(fileIndexHTML), "UTF-8");

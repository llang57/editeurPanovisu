77 public String chat(String userMessage, String systemMessage, double temperature, int maxTokens) throws Exception {
78 if (apiKey == null || apiKey.isEmpty()) {
79 throw new IllegalStateException("Clé API OpenRouter non configurée");
80 }
81
82 URI uri = URI.create(API_BASE_URL);
83 HttpURLConnection conn = (HttpURLConnection) uri.toURL().openConnection();
84
85 try {
86 // Configuration de la requête
87 conn.setRequestMethod("POST");
88 conn.setRequestProperty("Authorization", "Bearer " + apiKey);
89 conn.setRequestProperty("Content-Type", "application/json");
90 conn.setRequestProperty("HTTP-Referer", "https://github.com/panovisu/editeur");
91 conn.setRequestProperty("X-Title", "EditeurPanovisu");
92 conn.setDoOutput(true);
93
94 // Construction du corps de la requête
95 StringBuilder messagesJson = new StringBuilder();
96 messagesJson.append("[");
97
98 // Message système si présent
99 if (systemMessage != null && !systemMessage.isEmpty()) {
100 messagesJson.append("{\"role\": \"system\", \"content\": \"")
101 .append(escapeJson(systemMessage))
102 .append("\"},");
103 }
104
105 // Message utilisateur
106 messagesJson.append("{\"role\": \"user\", \"content\": \"")
107 .append(escapeJson(userMessage))
108 .append("\"}]");
109
110 String jsonInput = String.format(
111 "{\"model\": \"%s\", \"messages\": %s, \"temperature\": %.2f, \"max_tokens\": %d}",
112 modelName, messagesJson, temperature, maxTokens
113 );
114
115 // Envoi de la requête
116 try (OutputStream os = conn.getOutputStream()) {
117 byte[] input = jsonInput.getBytes(StandardCharsets.UTF_8);
118 os.write(input, 0, input.length);
119 }
120
121 // Lecture de la réponse
122 int responseCode = conn.getResponseCode();
123
124 if (responseCode == HttpURLConnection.HTTP_OK) {
125 try (BufferedReader br = new BufferedReader(
126 new InputStreamReader(conn.getInputStream(), StandardCharsets.UTF_8))) {
127 StringBuilder response = new StringBuilder();
128 String line;
129 while ((line = br.readLine()) != null) {
130 response.append(line);
131 }
132
133 // Extraction du message de la réponse
134 return extractMessageContent(response.toString());
135 }
136 } else {
137 try (BufferedReader br = new BufferedReader(
138 new InputStreamReader(conn.getErrorStream(), StandardCharsets.UTF_8))) {
139 StringBuilder error = new StringBuilder();
140 String line;
141 while ((line = br.readLine()) != null) {
142 error.append(line);
143 }
144 throw new Exception("Erreur API OpenRouter (" + responseCode + "): " + error);
145 }
146 }
147 } finally {
148 conn.disconnect();
149 }
150 }

220 private String extractMessageContent(String jsonResponse) {
221 // Parsing simple - pour un parsing robuste, utiliser une lib JSON
222 String marker = "\"content\":";
223 int startIdx = jsonResponse.indexOf(marker);
224 if (startIdx == -1) {
225 return jsonResponse; // Retourner la réponse brute si format non reconnu
226 }
227
228 startIdx += marker.length();
229 // Trouver le début de la chaîne
230 while (startIdx < jsonResponse.length() && jsonResponse.charAt(startIdx) != '"') {
231 startIdx++;
232 }
233 startIdx++; // Passer le guillemet
234
235 // Trouver la fin de la chaîne (en gérant les échappements)
236 int endIdx = startIdx;
237 while (endIdx < jsonResponse.length()) {
238 if (jsonResponse.charAt(endIdx) == '"' && jsonResponse.charAt(endIdx - 1) != '\\') {
239 break;
240 }
241 endIdx++;
242 }
243
244 String content = jsonResponse.substring(startIdx, endIdx);
245
246 // Déséchapper les caractères
247 return content.replace("\\n", "\n")
248 .replace("\\r", "\r")
249 .replace("\\t", "\t")
250 .replace("\\\"", "\"")
251 .replace("\\\\", "\\");
252 }

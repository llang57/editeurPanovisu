53 private void initializeGPU() {
54 try {
55 // Vérifier si OpenCL est disponible
56 try {
57 // Tenter d'accéder à la classe CL pour déclencher le chargement de la bibliothèque native
58 Class.forName("org.jocl.CL");
59 } catch (UnsatisfiedLinkError | ExceptionInInitializerError e) {
60 System.out.println("⚠️ OpenCL non disponible sur ce système - Mode CPU activé");
61 System.out.println(" Détails: " + e.getMessage());
62 gpuAvailable = false;
63 return;
64 } catch (ClassNotFoundException e) {
65 System.out.println("⚠️ Bibliothèque JOCL non trouvée - Mode CPU activé");
66 gpuAvailable = false;
67 return;
68 }
69
70 // Activer les exceptions OpenCL
71 CL.setExceptionsEnabled(true);
72
73 // 1. Récupérer les plateformes OpenCL
74 int[] numPlatforms = new int[1];
75 clGetPlatformIDs(0, null, numPlatforms);
76
77 if (numPlatforms[0] == 0) {
78 System.out.println("⚠️ Aucune plateforme OpenCL détectée");
79 return;
80 }
81
82 cl_platform_id[] platforms = new cl_platform_id[numPlatforms[0]];
83 clGetPlatformIDs(platforms.length, platforms, null);
84
85 // 2. Chercher un GPU
86 for (cl_platform_id p : platforms) {
87 // Récupérer le nom de la plateforme
88 long[] size = new long[1];
89 clGetPlatformInfo(p, CL_PLATFORM_NAME, 0, null, size);
90 byte[] buffer = new byte[(int)size[0]];
91 clGetPlatformInfo(p, CL_PLATFORM_NAME, buffer.length, Pointer.to(buffer), null);
92 String pName = new String(buffer, 0, buffer.length - 1);
93
94 // Chercher des GPU sur cette plateforme
95 int[] numDevices = new int[1];
96 int result = clGetDeviceIDs(p, CL_DEVICE_TYPE_GPU, 0, null, numDevices);
97
98 if (result == CL_SUCCESS && numDevices[0] > 0) {
99 cl_device_id[] devices = new cl_device_id[numDevices[0]];
100 clGetDeviceIDs(p, CL_DEVICE_TYPE_GPU, devices.length, devices, null);
101
102 // Prendre le premier GPU disponible
103 platform = p;
104 device = devices[0];
105 platformName = pName;
106
107 // Récupérer les infos du GPU
108 retrieveDeviceInfo();
109
110 // 3. Créer le contexte OpenCL
111 cl_context_properties contextProperties = new cl_context_properties();
112 contextProperties.addProperty(CL_CONTEXT_PLATFORM, platform);
113 context = clCreateContext(
114 contextProperties,
115 1,
116 new cl_device_id[]{device},
117 null,
118 null,
119 null
120 );
121
122 // 4. Créer la file de commandes
123 cl_queue_properties properties = new cl_queue_properties();
124 commandQueue = clCreateCommandQueueWithProperties(context, device, properties, null);
125
126 gpuAvailable = true;
127 System.out.println("✅ GPU initialisé avec succès");
128 System.out.println(" 📍 Plateforme: " + platformName);
129 System.out.println(" 🎮 GPU: " + deviceName);
130 System.out.println(" 💾 Mémoire: " + (deviceMemory / 1024 / 1024) + " MB");
131 System.out.println(" ⚡ Unités de calcul: " + computeUnits);
132 System.out.println(" 👥 Taille max workgroup: " + maxWorkGroupSize);
133 break;
134 }
135 }
136
137 if (!gpuAvailable) {
138 System.out.println("⚠️ Aucun GPU OpenCL détecté, utilisation du CPU");
139 }
140
141 } catch (UnsatisfiedLinkError | ExceptionInInitializerError e) {
142 System.out.println("⚠️ Erreur de chargement de la bibliothèque OpenCL: " + e.getMessage());
143 System.out.println(" L'application continuera en mode CPU");
144 gpuAvailable = false;
145 } catch (Exception e) {
146 System.err.println("❌ Erreur lors de l'initialisation OpenCL: " + e.getMessage());
147 gpuAvailable = false;
148 }
149 }

154 private void retrieveDeviceInfo() {
155 try {
156 // Nom du device
157 long[] size = new long[1];
158 clGetDeviceInfo(device, CL_DEVICE_NAME, 0, null, size);
159 byte[] buffer = new byte[(int)size[0]];
160 clGetDeviceInfo(device, CL_DEVICE_NAME, buffer.length, Pointer.to(buffer), null);
161 deviceName = new String(buffer, 0, buffer.length - 1);
162
163 // Mémoire globale
164 long[] mem = new long[1];
165 clGetDeviceInfo(device, CL_DEVICE_GLOBAL_MEM_SIZE, Sizeof.cl_long, Pointer.to(mem), null);
166 deviceMemory = mem[0];
167
168 // Unités de calcul
169 int[] units = new int[1];
170 clGetDeviceInfo(device, CL_DEVICE_MAX_COMPUTE_UNITS, Sizeof.cl_int, Pointer.to(units), null);
171 computeUnits = units[0];
172
173 // Taille max workgroup
174 long[] workgroup = new long[1];
175 clGetDeviceInfo(device, CL_DEVICE_MAX_WORK_GROUP_SIZE, Sizeof.size_t, Pointer.to(workgroup), null);
176 maxWorkGroupSize = workgroup[0];
177
178 } catch (Exception e) {
179 System.err.println("⚠️ Erreur lors de la récupération des infos GPU: " + e.getMessage());
180 }

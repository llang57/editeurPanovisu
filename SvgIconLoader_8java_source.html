144 public static ImageView loadSvgIcon(String iconName, int width, int height, Color color) {
145 // Si aucune couleur spécifiée, détecter automatiquement selon le thème
146 if (color == null) {
147 color = getThemeColor();
148 }
149
150 // Créer une clé unique pour le cache
151 String cacheKey = iconName + "_" + width + "x" + height + "_" + color.toString();
152
153 // Vérifier si l'icône est déjà en cache
154 if (iconCache.containsKey(cacheKey)) {
155 return new ImageView(iconCache.get(cacheKey));
156 }
157
158 try {
159 String svgPath = baseAppPath + File.separator + SVG_PATH + iconName + ".svg";
160 File svgFile = new File(svgPath);
161
162 // Si le SVG n'existe pas, essayer de charger le PNG équivalent
163 if (!svgFile.exists()) {
164 return loadFallbackPng(iconName, width);
165 }
166
167 // Lire le contenu du SVG
168 String svgContent = new String(Files.readAllBytes(Paths.get(svgPath)), "UTF-8");
169
170 // Remplacer les couleurs blanches par currentColor pour qu'elles prennent la couleur du thème
171 svgContent = svgContent.replaceAll("fill\\s*=\\s*['\"]white['\"]", "fill=\"currentColor\"");
172 svgContent = svgContent.replaceAll("fill\\s*=\\s*['\"]#[Ff]{6}['\"]", "fill=\"currentColor\"");
173 svgContent = svgContent.replaceAll("fill\\s*=\\s*['\"]#[Ff]{3}['\"]", "fill=\"currentColor\"");
174 svgContent = svgContent.replaceAll("fill\\s*:\\s*white", "fill:currentColor");
175 svgContent = svgContent.replaceAll("fill\\s*:\\s*#[Ff]{6}", "fill:currentColor");
176 svgContent = svgContent.replaceAll("fill\\s*:\\s*#[Ff]{3}", "fill:currentColor");
177
178 // Remplacer currentColor par la couleur spécifiée du thème
179 String hexColor = String.format("#%02X%02X%02X",
180 (int)(color.getRed() * 255),
181 (int)(color.getGreen() * 255),
182 (int)(color.getBlue() * 255));
183 svgContent = svgContent.replace("currentColor", hexColor);
184
185 // Convertir le SVG en PNG avec Batik
186 BufferedImage bufferedImage = convertSvgToPng(svgContent, width, height);
187
188 // Convertir BufferedImage en JavaFX Image
189 Image fxImage = SwingFXUtils.toFXImage(bufferedImage, null);
190
191 // Mettre en cache
192 iconCache.put(cacheKey, fxImage);
193
194 return new ImageView(fxImage);
195
196 } catch (Exception e) {
197 System.err.println("Erreur lors du chargement de l'icône SVG: " + iconName);
198 e.printStackTrace();
199 return loadFallbackPng(iconName, width);
200 }

221 private static BufferedImage convertSvgToPng(String svgContent, int width, int height) throws Exception {
222 // Créer le transcoder PNG
223 PNGTranscoder transcoder = new PNGTranscoder();
224
225 // Définir la taille de sortie
226 transcoder.addTranscodingHint(PNGTranscoder.KEY_WIDTH, (float) width);
227 transcoder.addTranscodingHint(PNGTranscoder.KEY_HEIGHT, (float) height);
228
229 // Créer un flux pour le SVG
230 ByteArrayInputStream inputStream = new ByteArrayInputStream(svgContent.getBytes("UTF-8"));
231 TranscoderInput input = new TranscoderInput(inputStream);
232
233 // Créer un flux de sortie pour le PNG
234 ByteArrayOutputStream outputStream = new ByteArrayOutputStream();
235 TranscoderOutput output = new TranscoderOutput(outputStream);
236
237 // Effectuer la conversion
238 transcoder.transcode(input, output);
239
240 // Convertir le flux en BufferedImage
241 ByteArrayInputStream pngInputStream = new ByteArrayInputStream(outputStream.toByteArray());
242 return javax.imageio.ImageIO.read(pngInputStream);
243 }

149 public static Image imgTransformationImage(Image imgRect, int iRapport) {
150 int iLargeur = (int) imgRect.getWidth() / iRapport;
151 int iHauteur = iLargeur / 2 / iRapport;
152 WritableImage imgMercator = new WritableImage(iLargeur, iHauteur);
153 PixelReader prRect = imgRect.getPixelReader();
154 PixelWriter pwMercator = imgMercator.getPixelWriter();
155 for (int i = 0; i < iLargeur; i++) {
156 for (int j = 0; j < iHauteur; j++) {
157 //double phi = 2 * Math.atan(Math.exp(2.d * Math.PI / largeur * (largeur / 4.d - (double) j))) - Math.PI / 2.d;
158 double phi = Math.asin(2.d * (iHauteur / 2.d - j) / iHauteur);
159 int y2 = (int) (iHauteur * iRapport * (0.5d - phi / Math.PI));
160 if (y2 >= iHauteur * iRapport) {
161 y2 = iHauteur * iRapport - 1;
162 }
163 Color clPixel = prRect.getColor(i * iRapport, y2 * iRapport);
164 pwMercator.setColor(i, j, clPixel);
165 }
166 }
167 return imgMercator;
168 }

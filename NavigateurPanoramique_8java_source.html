222 public static Image imgTransformationImage(Image imgRect, int iRapport) {
223 int iLargeur = (int) imgRect.getWidth() / iRapport;
224 int iHauteur = iLargeur / 2 / iRapport;
225 WritableImage imgMercator = new WritableImage(iLargeur, iHauteur);
226 PixelReader prRect = imgRect.getPixelReader();
227 PixelWriter pwMercator = imgMercator.getPixelWriter();
228 for (int i = 0; i < iLargeur; i++) {
229 for (int j = 0; j < iHauteur; j++) {
230 //double phi = 2 * Math.atan(Math.exp(2.d * Math.PI / largeur * (largeur / 4.d - (double) j))) - Math.PI / 2.d;
231 double phi = Math.asin(2.d * (iHauteur / 2.d - j) / iHauteur);
232 int y2 = (int) (iHauteur * iRapport * (0.5d - phi / Math.PI));
233 if (y2 >= iHauteur * iRapport) {
234 y2 = iHauteur * iRapport - 1;
235 }
236 Color clPixel = prRect.getColor(i * iRapport, y2 * iRapport);
237 pwMercator.setColor(i, j, clPixel);
238 }
239 }
240 return imgMercator;
241 }

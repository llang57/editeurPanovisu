85 private static ByteBuffer imageToByteBuffer(Image image) {
86 int width = (int) image.getWidth();
87 int height = (int) image.getHeight();
88
89 ByteBuffer buffer = ByteBuffer.allocateDirect(width * height * 4);
90 PixelReader reader = image.getPixelReader();
91
92 for (int y = 0; y < height; y++) {
93 for (int x = 0; x < width; x++) {
94 Color color = reader.getColor(x, y);
95 buffer.put((byte) (color.getRed() * 255));
96 buffer.put((byte) (color.getGreen() * 255));
97 buffer.put((byte) (color.getBlue() * 255));
98 buffer.put((byte) (color.getOpacity() * 255));
99 }
100 }

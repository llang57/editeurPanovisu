198 private static String escapeJavaStyleString(String strEntree,
199 boolean bEscapeSingleQuote, boolean escapeForwardSlash) {
200 StringBuilder sbSortie = new StringBuilder("");
201 if (strEntree == null) {
202 return null;
203 }
204 int sz;
205 sz = strEntree.length();
206 for (int i = 0; i < sz; i++) {
207 char chEntree = strEntree.charAt(i);
208 if (chEntree > 0xfff) {
209 sbSortie.append("\\u").append(hex(chEntree));
210 } else if (chEntree > 0xff) {
211 sbSortie.append("\\u0").append(hex(chEntree));
212 } else if (chEntree > 0x7f) {
213 sbSortie.append("\\u00").append(hex(chEntree));
214 } else if (chEntree < 32) {
215 switch (chEntree) {
216 case '\b':
217 sbSortie.append('\\');
218 sbSortie.append('b');
219 break;
220 case '\n':
221 sbSortie.append('\\');
222 sbSortie.append('n');
223 break;
224 case '\t':
225 sbSortie.append('\\');
226 sbSortie.append('t');
227 break;
228 case '\f':
229 sbSortie.append('\\');
230 sbSortie.append('f');
231 break;
232 case '\r':
233 sbSortie.append('\\');
234 sbSortie.append('r');
235 break;
236 default:
237 if (chEntree > 0xf) {
238 sbSortie.append("\\u00").append(hex(chEntree));
239 } else {
240 sbSortie.append("\\u000").append(hex(chEntree));
241 }
242 break;
243 }
244 } else {
245 switch (chEntree) {
246 case '\'':
247 if (bEscapeSingleQuote) {
248 sbSortie.append('\\');
249 }
250 sbSortie.append('\'');
251 break;
252 case '"':
253 sbSortie.append('\\');
254 sbSortie.append('"');
255 break;
256 case '\\':
257 sbSortie.append('\\');
258 sbSortie.append('\\');
259 break;
260 case '/':
261 if (escapeForwardSlash) {
262 sbSortie.append('\\');
263 }
264 sbSortie.append('/');
265 break;
266 default:
267 sbSortie.append(chEntree);
268 break;
269 }
270 }
271 }
272 return sbSortie.toString();
273 }

1904 private String formaterHTML(String html) {
1905 StringBuilder formatted = new StringBuilder();
1906 int indentLevel = 0;
1907 String indent = " "; // 2 espaces par niveau
1908
1909 // Supprimer les espaces inutiles entre les balises
1910 html = html.replaceAll(">\\s+<", "><");
1911
1912 // Découper en balises
1913 String[] parts = html.split("(?=<)|(?<=>)");
1914
1915 for (String part : parts) {
1916 if (part.trim().isEmpty()) continue;
1917
1918 // Balise fermante
1919 if (part.startsWith("</")) {
1920 indentLevel = Math.max(0, indentLevel - 1);
1921 formatted.append("\n");
1922 for (int i = 0; i < indentLevel; i++) {
1923 formatted.append(indent);
1924 }
1925 formatted.append(part);
1926 }
1927 // Balise auto-fermante ou balise qui ne contient pas d'enfants
1928 else if (part.matches("<[^>]*/\\s*>") || part.matches("<(img|br|hr|input|meta|link)[^>]*>")) {
1929 formatted.append("\n");
1930 for (int i = 0; i < indentLevel; i++) {
1931 formatted.append(indent);
1932 }
1933 formatted.append(part);
1934 }
1935 // Balise ouvrante
1936 else if (part.startsWith("<") && !part.startsWith("<!--")) {
1937 formatted.append("\n");
1938 for (int i = 0; i < indentLevel; i++) {
1939 formatted.append(indent);
1940 }
1941 formatted.append(part);
1942 // Pas d'indentation pour les balises inline courtes
1943 if (!part.matches("<(span|a|strong|em|b|i|u)[^>]*>")) {
1944 indentLevel++;
1945 }
1946 }
1947 // Commentaire
1948 else if (part.startsWith("<!--")) {
1949 formatted.append("\n");
1950 for (int i = 0; i < indentLevel; i++) {
1951 formatted.append(indent);
1952 }
1953 formatted.append(part);
1954 }
1955 // Texte
1956 else {
1957 String text = part.trim();
1958 if (!text.isEmpty()) {
1959 // Si le texte est long, on le met sur sa propre ligne
1960 if (text.length() > 60) {
1961 formatted.append("\n");
1962 for (int i = 0; i < indentLevel; i++) {
1963 formatted.append(indent);
1964 }
1965 }
1966 formatted.append(text);
1967 }
1968 }
1969 }
1970
1971 return formatted.toString().trim();
1972 }

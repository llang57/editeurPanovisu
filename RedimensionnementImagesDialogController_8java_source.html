815 private void ouvrirFileChooser() {
816 FileChooser fc = new FileChooser();
817 fc.setTitle(rbLocalisation.getString("selectionnerImages"));
818
819 // Utiliser le dernier répertoire d'images mémorisé
820 String dernierRepertoire = EditeurPanovisu.getStrDernierRepertoireImages();
821 if (dernierRepertoire != null && !dernierRepertoire.isEmpty()) {
822 File repImages = new File(dernierRepertoire);
823 if (repImages.exists() && repImages.isDirectory()) {
824 fc.setInitialDirectory(repImages);
825 } else {
826 fc.setInitialDirectory(new File(strRepertoireDefaut));
827 }
828 } else {
829 fc.setInitialDirectory(new File(strRepertoireDefaut));
830 }
831
832 // Filtres d'extensions
833 FileChooser.ExtensionFilter filtreImages = new FileChooser.ExtensionFilter(
834 rbLocalisation.getString("fichiersImages"),
835 "*.jpg", "*.jpeg", "*.png", "*.bmp", "*.gif", "*.webp"
836 );
837 fc.getExtensionFilters().add(filtreImages);

1338 sauvegarderImage(imageRedimensionnee, fichierSortie, format);

1362 private BufferedImage redimensionnerImage(BufferedImage imageSource, int largeur, int hauteur) {
1363 // Vérifier si le GPU est disponible
1364 GPUManager gpu = GPUManager.getInstance();
1365 boolean gpuDisponible = gpu.isGPUAvailable() && gpu.isGPUEnabled();
1366
1367 if (gpuDisponible) {
1368 try {
1369 Logger.getLogger(RedimensionnementImagesDialogController.class.getName()).log(
1370 Level.INFO,
1371 String.format("🎨 Début redimensionnement GPU: %dx%d → %dx%d",
1372 imageSource.getWidth(), imageSource.getHeight(), largeur, hauteur
1373 )
1374 );
1375
1376 // L'image est déjà normalisée en RGB dans traiterImage()
1377 // Conversion BufferedImage → JavaFX Image
1378 Logger.getLogger(RedimensionnementImagesDialogController.class.getName()).log(
1379 Level.INFO,
1380 "🔄 Conversion BufferedImage → JavaFX Image..."
1381 );
1382
1383 Image fxImage = SwingFXUtils.toFXImage(imageSource, null);
1384
1385 Logger.getLogger(RedimensionnementImagesDialogController.class.getName()).log(
1386 Level.INFO,
1387 "✅ Conversion JavaFX réussie"
1388 );
1389
1390 // Calcul du facteur de redimensionnement
1391 double facteurLargeur = (double) largeur / imageSource.getWidth();
1392 double facteurHauteur = (double) hauteur / imageSource.getHeight();
1393 double facteur = Math.max(facteurLargeur, facteurHauteur);
1394
1395 // Sélection de la méthode d'interpolation
1396 InterpolationMethod methode;
1397
1398 if (facteur >= 2.0) {
1399 // Agrandissement ×2+ → Lanczos3 (meilleure qualité pour upscaling)
1400 methode = InterpolationMethod.LANCZOS3;
1401 } else if (facteur <= 0.25) {
1402 // Réduction ÷4+ → Lanczos3 (meilleur anti-aliasing)
1403 methode = InterpolationMethod.LANCZOS3;
1404 } else {
1405 // Cas général → Bicubique (optimal)
1406 methode = InterpolationMethod.BICUBIC;
1407 }
1408
1409 // Redimensionnement GPU
1410 Image fxImageRedim = ImageResizeGPU.resizeAuto(fxImage, largeur, hauteur, methode);
1411
1412 Logger.getLogger(RedimensionnementImagesDialogController.class.getName()).log(
1413 Level.INFO,
1414 "🔄 Conversion JavaFX Image → BufferedImage..."
1415 );
1416
1417 // Conversion JavaFX Image → BufferedImage avec création explicite RGB
1418 // Créer un BufferedImage RGB de destination
1419 BufferedImage resultat = new BufferedImage(
1420 (int) fxImageRedim.getWidth(),
1421 (int) fxImageRedim.getHeight(),
1422 BufferedImage.TYPE_INT_RGB // Force RGB pour éviter "Bogus colorspace"
1423 );
1424
1425 // Convertir en passant le BufferedImage de destination
1426 SwingFXUtils.fromFXImage(fxImageRedim, resultat);
1427
1428 Logger.getLogger(RedimensionnementImagesDialogController.class.getName()).log(
1429 Level.INFO,
1430 "✅ Conversion BufferedImage réussie"
1431 );
1432
1433 // Log pour debug
1434 Logger.getLogger(RedimensionnementImagesDialogController.class.getName()).log(
1435 Level.INFO,
1436 String.format("🎨 Redimensionnement GPU %s : %dx%d → %dx%d (facteur: %.2f×, méthode: %s)",
1437 facteur >= 1.0 ? "Agrandissement" : "Réduction",
1438 imageSource.getWidth(), imageSource.getHeight(),
1439 largeur, hauteur,
1440 facteur,
1441 methode.name()
1442 )
1443 );
1444
1445 return resultat;
1446
1447 } catch (Exception e) {
1448 // En cas d'erreur GPU, basculer sur CPU
1449 Logger.getLogger(RedimensionnementImagesDialogController.class.getName()).log(
1450 Level.WARNING,
1451 "⚠️ Erreur GPU, bascule sur CPU : " + e.getMessage()
1452 );
1453 }
1454 }
1455
1456 // Fallback CPU : Graphics2D avec Bilinéaire
1457 BufferedImage imageRedimensionnee = new BufferedImage(largeur, hauteur, BufferedImage.TYPE_INT_RGB);
1458 Graphics2D g2d = imageRedimensionnee.createGraphics();
1459
1460 // Configuration CPU optimisée (Bilinéaire rapide)
1461 g2d.setRenderingHint(RenderingHints.KEY_INTERPOLATION, RenderingHints.VALUE_INTERPOLATION_BILINEAR);
1462 g2d.setRenderingHint(RenderingHints.KEY_RENDERING, RenderingHints.VALUE_RENDER_SPEED);
1463 g2d.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_ON);
1464
1465 g2d.drawImage(imageSource, 0, 0, largeur, hauteur, null);
1466 g2d.dispose();
1467
1468 Logger.getLogger(RedimensionnementImagesDialogController.class.getName()).log(
1469 Level.INFO,
1470 String.format("💻 Redimensionnement CPU : %dx%d → %dx%d (Bilinéaire)",
1471 imageSource.getWidth(), imageSource.getHeight(),
1472 largeur, hauteur
1473 )
1474 );
1475
1476 return imageRedimensionnee;
1477 }

152 public void chargerDocument(DocumentType type) {
153 try {
154 File file = new File(type.getChemin());
155 if (file.exists()) {
156 markdownViewer.afficherFichierMarkdown(webView, type.getChemin());
157 } else {
158 afficherErreur("Fichier non trouvé : " + type.getChemin());
159 }
160 } catch (IOException e) {
161 afficherErreur("Erreur lors de la lecture du fichier : " + e.getMessage());
162 }
163 }

168 private void ouvrirFichierMarkdown() {
169 FileChooser fileChooser = new FileChooser();
170 fileChooser.setTitle("Ouvrir un fichier Markdown");
171 fileChooser.getExtensionFilters().addAll(
172 new FileChooser.ExtensionFilter("Fichiers Markdown", "*.md", "*.markdown"),
173 new FileChooser.ExtensionFilter("Tous les fichiers", "*.*")
174 );
175
176 File selectedFile = fileChooser.showOpenDialog(this);
177 if (selectedFile != null) {
178 try {
179 markdownViewer.afficherFichierMarkdown(webView, selectedFile.getAbsolutePath());
180 } catch (IOException e) {
181 afficherErreur("Erreur lors de la lecture du fichier : " + e.getMessage());
182 }
183 }

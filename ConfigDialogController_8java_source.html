416 private void saveModelsPreferences() throws IOException {
417 File filePreferences = new File(EditeurPanovisu.fileRepertConfig.getAbsolutePath() +
418 File.separator + "preferences.cfg");
419
420 // Lire les pr√©f√©rences existantes
421 StringBuilder existingContent = new StringBuilder();
422 if (filePreferences.exists()) {
423 try (BufferedReader reader = new BufferedReader(
424 new InputStreamReader(new FileInputStream(filePreferences), "UTF-8"))) {
425 String line;
426 while ((line = reader.readLine()) != null) {
427 // Ignorer les anciennes lignes de mod√®les pour √©viter les doublons
428 if (!line.startsWith("openrouterModel=") && !line.startsWith("ollamaModel=")) {
429 existingContent.append(line).append("\n");
430 }
431 }
432 }
433 }
434
435 // Ajouter les nouveaux param√®tres de mod√®les
436 String openRouterModel = OllamaService.getOpenRouterModel();
437 String ollamaModel = OllamaService.getOllamaModel();
438
439 if (openRouterModel != null && !openRouterModel.isEmpty()) {
440 existingContent.append("openrouterModel=").append(openRouterModel).append("\n");
441 }
442 if (ollamaModel != null && !ollamaModel.isEmpty()) {
443 existingContent.append("ollamaModel=").append(ollamaModel).append("\n");
444 }
445
446 // R√©√©crire le fichier
447 filePreferences.setWritable(true);
448 try (OutputStreamWriter writer = new OutputStreamWriter(
449 new FileOutputStream(filePreferences), "UTF-8");

451 bw.write(existingContent.toString());
452 }

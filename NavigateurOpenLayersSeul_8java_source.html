196 public void afficheCartesOpenlayer() {
197 getApChoixCartographie().getChildren().clear();
198 String nomsCartes = recupereCartographiesOpenLayers();
199 strCartesOpenLayers = nomsCartes.split("\\|");
200 for (int i = 0; i < strCartesOpenLayers.length; i++) {
201 boolean bCarteChoisie = false;
202 if (strCartesOpenLayers[i].substring(0, 1).equals("*")) {
203 strCartesOpenLayers[i] = strCartesOpenLayers[i].substring(1, strCartesOpenLayers[i].length());
204 bCarteChoisie = true;
205 }
206 RadioButton rbCarto = new RadioButton(strCartesOpenLayers[i]);
207 rbCarto.setLayoutX(10);
208 rbCarto.setLayoutY(i * 30);
209 rbCarto.setUserData(strCartesOpenLayers[i]);
210 rbCarto.setSelected(bCarteChoisie);
211 rbCarto.setToggleGroup(tgCartesOpenLayers);
212 getApChoixCartographie().getChildren().add(rbCarto);
213 }
214 getApChoixCartographie().setPrefHeight(strCartesOpenLayers.length * 30 + 10);
215 }

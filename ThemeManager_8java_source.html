167 public static void applyTheme(Scene scene, Theme theme) {
168 if (scene == null) {
169 return;
170 }
171
172 // Nettoyer les anciens stylesheets
173 scene.getStylesheets().clear();
174
175 // Nettoyer les classes CSS du root
176 cleanRootClasses(scene);
177
178 // Appliquer le thème selon son type
179 switch (theme.getProvider()) {
180 case CUSTOM:
181 applyCustomTheme(scene, theme);
182 break;
183 case MATERIALFX:
184 applyMaterialFXTheme(scene, theme);
185 break;
186 case FLATLAF:
187 applyFlatLafTheme(scene, theme);
188 break;
189 case ATLANTAFX:
190 default:
191 applyAtlantaFXTheme(scene, theme);
192 break;
193 }
194
195 // Ajouter les styles personnalisés de l'application
196 addCustomStyles(scene, theme);
197
198 currentTheme = theme;
199 saveThemePreference(theme);
200
201 // Vider le cache des icônes SVG pour forcer le rechargement avec la nouvelle couleur
202 try {
203 Class<?> svgLoaderClass = Class.forName("editeurpanovisu.util.SvgIconLoader");
204 java.lang.reflect.Method clearCacheMethod = svgLoaderClass.getMethod("clearCache");
205 clearCacheMethod.invoke(null);
206 System.out.println(" ↳ Cache des icônes SVG vidé pour le nouveau thème");
207
208 // Recharger les icônes de la barre d'outils avec les nouvelles couleurs
209 Class<?> editeurClass = Class.forName("editeurpanovisu.EditeurPanovisu");
210 java.lang.reflect.Method rechargerIconesMethod = editeurClass.getDeclaredMethod("rechargerIcones");
211 rechargerIconesMethod.setAccessible(true);
212 rechargerIconesMethod.invoke(null);
213 System.out.println(" ↳ Icônes de la barre d'outils rechargées");
214 } catch (Exception e) {
215 // Si SvgIconLoader ou rechargerIcones n'est pas disponible, ce n'est pas grave
216 System.out.println(" ⚠ Impossible de recharger les icônes : " + e.getMessage());
217 }
218 }

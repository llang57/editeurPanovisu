166 private static String formatBuildNumber(int buildNumber) {
167 String buildStr = String.valueOf(buildNumber);
168 if (buildStr.length() <= 3) {
169 return buildStr;
170 }
171
172 // Séparer les milliers avec \u00a0 (espace insécable)
173 StringBuilder formatted = new StringBuilder();
174 int len = buildStr.length();
175 for (int i = 0; i < len; i++) {
176 if (i > 0 && (len - i) % 3 == 0) {
177 formatted.append("\\u00a0");
178 }
179 formatted.append(buildStr.charAt(i));
180 }

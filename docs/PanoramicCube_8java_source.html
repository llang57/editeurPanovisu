97 Image resizedImage = resizeToEquirectangular(panoramicImage, 1500, 750);

99 // Convertir l'image équirectangulaire en 6 faces de cube
100 // Order equi2cube: [0]=Front, [1]=Behind, [2]=Right, [3]=Left, [4]=Top, [5]=Bottom
101 Image[] cubeFaces = TransformationsPanoramique.equi2cube(resizedImage, 500);
102
103 // Mapper les faces correctement avec selfIlluminationMap pour éliminer les ombres
104 // Notre ordre: [0]=FRONT, [1]=BACK, [2]=LEFT, [3]=RIGHT, [4]=TOP, [5]=BOTTOM
105 materials[FRONT].setSelfIlluminationMap(cubeFaces[0]); // Front -> Front
106 materials[BACK].setSelfIlluminationMap(cubeFaces[1]); // Behind -> Back
107 materials[LEFT].setSelfIlluminationMap(cubeFaces[3]); // Left -> Left
108 materials[RIGHT].setSelfIlluminationMap(cubeFaces[2]); // Right -> Right
109 materials[TOP].setSelfIlluminationMap(cubeFaces[4]); // Top -> Top
110 materials[BOTTOM].setSelfIlluminationMap(cubeFaces[5]); // Bottom -> Bottom
111
112 // Désactiver diffuse et spéculaire pour éliminer totalement les ombres
113 for (int i = 0; i < 6; i++) {
114 materials[i].setDiffuseMap(null);
115 materials[i].setSpecularMap(null);
116 materials[i].setDiffuseColor(Color.BLACK); // Pas de couleur diffuse
117 }
118 } catch (InterruptedException ex) {
119 // En cas d'interruption, appliquer l'image complète (fallback)
120 for (int i = 0; i < 6; i++) {
121 materials[i].setSelfIlluminationMap(panoramicImage);
122 materials[i].setDiffuseMap(null);
123 materials[i].setSpecularMap(null);
124 materials[i].setDiffuseColor(Color.BLACK);
125 }
126 }

137 private Image resizeToEquirectangular(Image source, int targetWidth, int targetHeight) {
138 int srcWidth = (int) source.getWidth();
139 int srcHeight = (int) source.getHeight();
140
141 // Si l'image est déjà à la bonne taille, la retourner directement
142 if (srcWidth == targetWidth && srcHeight == targetHeight) {
143 return source;
144 }
145
146 // Créer une nouvelle image redimensionnée
147 WritableImage resized = new WritableImage(targetWidth, targetHeight);
148 PixelReader reader = source.getPixelReader();
149 PixelWriter writer = resized.getPixelWriter();
150
151 // Redimensionnement par échantillonnage simple
152 double xRatio = (double) srcWidth / targetWidth;
153 double yRatio = (double) srcHeight / targetHeight;
154
155 for (int y = 0; y < targetHeight; y++) {
156 for (int x = 0; x < targetWidth; x++) {
157 int srcX = (int) (x * xRatio);
158 int srcY = (int) (y * yRatio);
159
160 // S'assurer qu'on ne dépasse pas les limites de l'image source
161 if (srcX >= srcWidth) srcX = srcWidth - 1;
162 if (srcY >= srcHeight) srcY = srcHeight - 1;
163
164 Color color = reader.getColor(srcX, srcY);
165 writer.setColor(x, y, color);
166 }
167 }
168
169 return resized;
170 }

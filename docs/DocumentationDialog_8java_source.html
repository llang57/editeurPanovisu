153 private void ouvrirFichierMarkdown() {
154 FileChooser fileChooser = new FileChooser();
155 fileChooser.setTitle("Ouvrir un fichier Markdown");
156 fileChooser.getExtensionFilters().addAll(
157 new FileChooser.ExtensionFilter("Fichiers Markdown", "*.md", "*.markdown"),
158 new FileChooser.ExtensionFilter("Tous les fichiers", "*.*")
159 );
160
161 File selectedFile = fileChooser.showOpenDialog(this);
162 if (selectedFile != null) {
163 try {
164 markdownViewer.afficherFichierMarkdown(webView, selectedFile.getAbsolutePath());
165 } catch (IOException e) {
166 afficherErreur("Erreur lors de la lecture du fichier : " + e.getMessage());
167 }

143 public static void applyTheme(Scene scene, Theme theme) {
144 if (scene == null) {
145 return;
146 }
147
148 // Nettoyer les anciens stylesheets
149 scene.getStylesheets().clear();
150
151 // Nettoyer les classes CSS du root
152 cleanRootClasses(scene);
153
154 // Appliquer le thème selon son type
155 switch (theme.getProvider()) {
156 case CUSTOM:
157 applyCustomTheme(scene, theme);
158 break;
159 case MATERIALFX:
160 applyMaterialFXTheme(scene, theme);
161 break;
162 case FLATLAF:
163 applyFlatLafTheme(scene, theme);
164 break;
165 case ATLANTAFX:
166 default:
167 applyAtlantaFXTheme(scene, theme);
168 break;
169 }
170
171 // Ajouter les styles personnalisés de l'application
172 addCustomStyles(scene, theme);
173
174 currentTheme = theme;
175 saveThemePreference(theme);
176
177 // Vider le cache des icônes SVG pour forcer le rechargement avec la nouvelle couleur
178 try {
179 Class<?> svgLoaderClass = Class.forName("editeurpanovisu.util.SvgIconLoader");
180 java.lang.reflect.Method clearCacheMethod = svgLoaderClass.getMethod("clearCache");
181 clearCacheMethod.invoke(null);
182 System.out.println(" ↳ Cache des icônes SVG vidé pour le nouveau thème");
183
184 // Recharger les icônes de la barre d'outils avec les nouvelles couleurs
185 Class<?> editeurClass = Class.forName("editeurpanovisu.EditeurPanovisu");
186 java.lang.reflect.Method rechargerIconesMethod = editeurClass.getDeclaredMethod("rechargerIcones");
187 rechargerIconesMethod.setAccessible(true);
188 rechargerIconesMethod.invoke(null);
189 System.out.println(" ↳ Icônes de la barre d'outils rechargées");
190 } catch (Exception e) {
191 // Si SvgIconLoader ou rechargerIcones n'est pas disponible, ce n'est pas grave
192 System.out.println(" ⚠ Impossible de recharger les icônes : " + e.getMessage());
193 }
194 }

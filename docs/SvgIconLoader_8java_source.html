60 private static Color getThemeColor() {
61 try {
62 // Utiliser ThemeManager pour détecter le thème actuel
63 Class<?> themeManagerClass = Class.forName("editeurpanovisu.ThemeManager");
64 java.lang.reflect.Method getCurrentThemeMethod = themeManagerClass.getMethod("getCurrentTheme");
65 Object currentTheme = getCurrentThemeMethod.invoke(null);
66
67 // Récupérer le nom du thème
68 java.lang.reflect.Method nameMethod = currentTheme.getClass().getMethod("name");
69 String themeName = (String) nameMethod.invoke(currentTheme);
70
71 // Couleurs spéciales pour Dracula et Darcula
72 if ("DRACULA".equals(themeName)) {
73 return Color.web("#BD93F9"); // Mauve Dracula caractéristique
74 } else if ("FLATLAF_DARCULA".equals(themeName)) {
75 return Color.web("#9876AA"); // Violet Darcula IntelliJ
76 }
77
78 // Vérifier si le thème est sombre
79 java.lang.reflect.Method isDarkMethod = currentTheme.getClass().getMethod("isDark");
80 boolean isDark = (Boolean) isDarkMethod.invoke(currentTheme);
81
82 // Thème sombre = icônes claires, thème clair = icônes foncées
83 if (isDark) {
84 return Color.web("#E0E0E0"); // Gris très clair pour thèmes sombres
85 } else {
86 return Color.web("#333333"); // Gris foncé pour thèmes clairs
87 }
88 } catch (Exception e) {
89 // Fallback sur l'ancienne méthode avec EditeurPanovisu.getStrStyleCSS()
90 try {
91 Class<?> editeurClass = Class.forName("editeurpanovisu.EditeurPanovisu");
92 java.lang.reflect.Method getStyleMethod = editeurClass.getMethod("getStrStyleCSS");
93 String theme = (String) getStyleMethod.invoke(null);
94
95 if ("fonce".equalsIgnoreCase(theme)) {
96 return Color.web("#E0E0E0");
97 } else {
98 return Color.web("#333333");
99 }
100 } catch (Exception ex) {
101 // En cas d'erreur totale, utiliser la couleur par défaut (thème clair)
102 System.err.println("Impossible de détecter le thème, utilisation de la couleur par défaut");
103 return Color.web("#333333");
104 }
105 }

116 public static ImageView loadSvgIcon(String iconName, int width, int height, Color color) {
117 // Si aucune couleur spécifiée, détecter automatiquement selon le thème
118 if (color == null) {
119 color = getThemeColor();
120 }
121
122 // Créer une clé unique pour le cache
123 String cacheKey = iconName + "_" + width + "x" + height + "_" + color.toString();
124
125 // Vérifier si l'icône est déjà en cache
126 if (iconCache.containsKey(cacheKey)) {
127 return new ImageView(iconCache.get(cacheKey));
128 }
129
130 try {
131 String svgPath = baseAppPath + File.separator + SVG_PATH + iconName + ".svg";
132 File svgFile = new File(svgPath);
133
134 // Si le SVG n'existe pas, essayer de charger le PNG équivalent
135 if (!svgFile.exists()) {
136 return loadFallbackPng(iconName, width);
137 }
138
139 // Lire le contenu du SVG
140 String svgContent = new String(Files.readAllBytes(Paths.get(svgPath)), "UTF-8");
141
142 // Remplacer currentColor par la couleur spécifiée
143 String hexColor = String.format("#%02X%02X%02X",
144 (int)(color.getRed() * 255),
145 (int)(color.getGreen() * 255),
146 (int)(color.getBlue() * 255));
147 svgContent = svgContent.replace("currentColor", hexColor);
148
149 // Remplacer également "white" par la couleur de fond (blanc pour thème clair)
150 // Note: ceci est une simplification, dans un vrai système il faudrait détecter le thème
151
152 // Convertir le SVG en PNG avec Batik
153 BufferedImage bufferedImage = convertSvgToPng(svgContent, width, height);
154
155 // Convertir BufferedImage en JavaFX Image
156 Image fxImage = SwingFXUtils.toFXImage(bufferedImage, null);
157
158 // Mettre en cache
159 iconCache.put(cacheKey, fxImage);
160
161 return new ImageView(fxImage);
162
163 } catch (Exception e) {
164 System.err.println("Erreur lors du chargement de l'icône SVG: " + iconName);
165 e.printStackTrace();
166 return loadFallbackPng(iconName, width);
167 }

188 private static BufferedImage convertSvgToPng(String svgContent, int width, int height) throws Exception {
189 // Créer le transcoder PNG
190 PNGTranscoder transcoder = new PNGTranscoder();
191
192 // Définir la taille de sortie
193 transcoder.addTranscodingHint(PNGTranscoder.KEY_WIDTH, (float) width);
194 transcoder.addTranscodingHint(PNGTranscoder.KEY_HEIGHT, (float) height);
195
196 // Créer un flux pour le SVG
197 ByteArrayInputStream inputStream = new ByteArrayInputStream(svgContent.getBytes("UTF-8"));
198 TranscoderInput input = new TranscoderInput(inputStream);
199
200 // Créer un flux de sortie pour le PNG
201 ByteArrayOutputStream outputStream = new ByteArrayOutputStream();
202 TranscoderOutput output = new TranscoderOutput(outputStream);
203
204 // Effectuer la conversion
205 transcoder.transcode(input, output);
206
207 // Convertir le flux en BufferedImage
208 ByteArrayInputStream pngInputStream = new ByteArrayInputStream(outputStream.toByteArray());
209 return javax.imageio.ImageIO.read(pngInputStream);
210 }

313 private Properties loadApiKeys() {
314 Properties props = new Properties();
315 File apiKeysFile = new File("api-keys.properties");
316
317 if (apiKeysFile.exists()) {
318 try (InputStreamReader reader = new InputStreamReader(new FileInputStream(apiKeysFile), "UTF-8")) {
319 props.load(reader);
320 System.out.println("📖 Clés API chargées depuis api-keys.properties");
321 } catch (IOException ex) {
322 Logger.getLogger(ConfigDialogController.class.getName()).log(Level.SEVERE, null, ex);
323 System.err.println("⚠️ Erreur lors du chargement de api-keys.properties");
324 }
325 } else {
326 System.out.println("⚠️ Fichier api-keys.properties non trouvé");
327 }
328
329 return props;
330 }

338 private void saveApiKeys(String locationIQKey, String huggingFaceKey, String openRouterKey) {
339 File apiKeysFile = new File("api-keys.properties");
340 Properties props = new Properties();
341
342 // Charger les propriétés existantes pour préserver les commentaires et autres clés
343 if (apiKeysFile.exists()) {
344 try (InputStreamReader reader = new InputStreamReader(new FileInputStream(apiKeysFile), "UTF-8")) {
345 props.load(reader);
346 } catch (IOException ex) {
347 Logger.getLogger(ConfigDialogController.class.getName()).log(Level.SEVERE, null, ex);
348 }
349 }
350
351 // Mettre à jour les clés
352 props.setProperty("locationiq.api.key", locationIQKey);
353 props.setProperty("huggingface.api.key", huggingFaceKey);
354 props.setProperty("openrouter.api.key", openRouterKey);
355
356 // Sauvegarder
357 try (OutputStreamWriter writer = new OutputStreamWriter(new FileOutputStream(apiKeysFile), "UTF-8")) {
358 props.store(writer, "Clés API - Modifié depuis les préférences");
359 System.out.println("✅ Clés API sauvegardées dans api-keys.properties");
360 } catch (IOException ex) {
361 Logger.getLogger(ConfigDialogController.class.getName()).log(Level.SEVERE, null, ex);
362 System.err.println("❌ Erreur lors de la sauvegarde de api-keys.properties");
363 }

369 private void saveModelsPreferences() throws IOException {
370 File filePreferences = new File(EditeurPanovisu.fileRepertConfig.getAbsolutePath() +
371 File.separator + "preferences.cfg");
372
373 // Lire les préférences existantes
374 StringBuilder existingContent = new StringBuilder();
375 if (filePreferences.exists()) {
376 try (BufferedReader reader = new BufferedReader(
377 new InputStreamReader(new FileInputStream(filePreferences), "UTF-8"))) {
378 String line;
379 while ((line = reader.readLine()) != null) {
380 // Ignorer les anciennes lignes de modèles pour éviter les doublons
381 if (!line.startsWith("openrouterModel=") && !line.startsWith("ollamaModel=")) {
382 existingContent.append(line).append("\n");
383 }
384 }
385 }
386 }
387
388 // Ajouter les nouveaux paramètres de modèles
389 String openRouterModel = OllamaService.getOpenRouterModel();
390 String ollamaModel = OllamaService.getOllamaModel();
391
392 if (openRouterModel != null && !openRouterModel.isEmpty()) {
393 existingContent.append("openrouterModel=").append(openRouterModel).append("\n");
394 }
395 if (ollamaModel != null && !ollamaModel.isEmpty()) {
396 existingContent.append("ollamaModel=").append(ollamaModel).append("\n");
397 }
398
399 // Réécrire le fichier
400 filePreferences.setWritable(true);
401 try (OutputStreamWriter writer = new OutputStreamWriter(
402 new FileOutputStream(filePreferences), "UTF-8");
403 BufferedWriter bw = new BufferedWriter(writer)) {
404 bw.write(existingContent.toString());
405 }
406
407 System.out.println("💾 Modèles IA sauvegardés dans preferences.cfg");
408 System.out.println(" - OpenRouter: " + openRouterModel);
409 System.out.println(" - Ollama: " + ollamaModel);
410 }

464 private static String extraireNomModele(String displayName) {
465 if (displayName == null || displayName.isEmpty()) {
466 return displayName;
467 }
468
469 // Mapping inverse (GPT-5 retiré)
470 if (displayName.contains("Claude Sonnet 4.5")) {
471 return "anthropic/claude-sonnet-4.5";
472 } else if (displayName.contains("Claude 3 Opus")) {
473 return "anthropic/claude-3-opus";
474 } else if (displayName.contains("Claude 3.5 Sonnet") && displayName.contains("oct 2024")) {
475 return "anthropic/claude-3.5-sonnet:20241022";
476 } else if (displayName.contains("Mistral Nemo") && displayName.contains("Mistral AI")) {
477 return "mistralai/mistral-nemo";
478 } else if (displayName.contains("Cogito v2 Preview")) {
479 return "deepcogito/cogito-v2-preview-deepseek-671b";
480 } else if (displayName.contains("GPT-OSS-120B")) {
481 return "openai/gpt-oss-120b";
482 } else if (displayName.contains("GPT-4 Turbo")) {
483 return "openai/gpt-4-turbo";
484 } else if (displayName.contains("Gemini Pro")) {
485 return "google/gemini-pro";
486 } else if (displayName.contains("Llama 3.1 8B")) {
487 return "meta-llama/llama-3.1-8b-instruct";
488 }
489 // Ollama
490 else if (displayName.contains("DeepSeek-R1")) {
491 // Extraire la version exacte (ex: deepseek-r1:70b)
492 if (displayName.contains(":")) {
493 // Si le nom complet avec version est dans displayName, le retourner
494 return displayName.replaceFirst("^[^a-z]+\\s*", "").split("\\s+")[0];
495 }
496 return "deepseek-r1:70b"; // Version par défaut
497 } else if (displayName.contains("Mistral Nemo")) {
498 return "mistral-nemo";
499 } else if (displayName.contains("Qwen 2.5")) {
500 return "qwen2.5";
501 } else if (displayName.contains("Llama 3.1") && !displayName.contains("8B")) {
502 return "llama3.1";
503 } else if (displayName.contains("Gemma 2")) {
504 return "gemma2";
505 }
506
507 // Si pas de correspondance, enlever juste l'emoji au début
508 return displayName.replaceFirst("^[^a-zA-Z0-9]+\\s*", "");
509 }

160 private static String escapeJavaStyleString(String strEntree,
161 boolean bEscapeSingleQuote, boolean escapeForwardSlash) {
162 StringBuilder sbSortie = new StringBuilder("");
163 if (strEntree == null) {
164 return null;
165 }
166 int sz;
167 sz = strEntree.length();
168 for (int i = 0; i < sz; i++) {
169 char chEntree = strEntree.charAt(i);
170 if (chEntree > 0xfff) {
171 sbSortie.append("\\u").append(hex(chEntree));
172 } else if (chEntree > 0xff) {
173 sbSortie.append("\\u0").append(hex(chEntree));
174 } else if (chEntree > 0x7f) {
175 sbSortie.append("\\u00").append(hex(chEntree));
176 } else if (chEntree < 32) {
177 switch (chEntree) {
178 case '\b':
179 sbSortie.append('\\');
180 sbSortie.append('b');
181 break;
182 case '\n':
183 sbSortie.append('\\');
184 sbSortie.append('n');
185 break;
186 case '\t':
187 sbSortie.append('\\');
188 sbSortie.append('t');
189 break;
190 case '\f':
191 sbSortie.append('\\');
192 sbSortie.append('f');
193 break;
194 case '\r':
195 sbSortie.append('\\');
196 sbSortie.append('r');
197 break;
198 default:
199 if (chEntree > 0xf) {
200 sbSortie.append("\\u00").append(hex(chEntree));
201 } else {
202 sbSortie.append("\\u000").append(hex(chEntree));
203 }
204 break;
205 }
206 } else {
207 switch (chEntree) {
208 case '\'':
209 if (bEscapeSingleQuote) {
210 sbSortie.append('\\');
211 }
212 sbSortie.append('\'');
213 break;
214 case '"':
215 sbSortie.append('\\');
216 sbSortie.append('"');
217 break;
218 case '\\':
219 sbSortie.append('\\');
220 sbSortie.append('\\');
221 break;
222 case '/':
223 if (escapeForwardSlash) {
224 sbSortie.append('\\');
225 }
226 sbSortie.append('/');
227 break;
228 default:
229 sbSortie.append(chEntree);
230 break;
231 }
232 }
233 }
234 return sbSortie.toString();
235 }

187 public void afficheCartesOpenlayer() {
188 getApChoixCartographie().getChildren().clear();
189 String nomsCartes = recupereCartographiesOpenLayers();
190 strCartesOpenLayers = nomsCartes.split("\\|");
191 for (int i = 0; i < strCartesOpenLayers.length; i++) {
192 boolean bCarteChoisie = false;
193 if (strCartesOpenLayers[i].substring(0, 1).equals("*")) {
194 strCartesOpenLayers[i] = strCartesOpenLayers[i].substring(1, strCartesOpenLayers[i].length());
195 bCarteChoisie = true;
196 }
197 RadioButton rbCarto = new RadioButton(strCartesOpenLayers[i]);
198 rbCarto.setLayoutX(10);
199 rbCarto.setLayoutY(i * 30);
200 rbCarto.setUserData(strCartesOpenLayers[i]);
201 rbCarto.setSelected(bCarteChoisie);
202 rbCarto.setToggleGroup(tgCartesOpenLayers);
203 getApChoixCartographie().getChildren().add(rbCarto);
204 }
205 getApChoixCartographie().setPrefHeight(strCartesOpenLayers.length * 30 + 10);
206 }

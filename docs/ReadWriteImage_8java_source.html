53 private static float[] calculeSharpenMatrix(float level) {
54 float[] normalMatrix = {
55 0.0f, 0.0f, 0.0f,
56 0.0f, 1.0f, 0.0f,
57 0.0f, 0.0f, 0.0f
58 };
59 float[] edgeMatrix = {
60 0.0f, -1.0f, 0.0f,
61 -1.0f, 4.0f, -1.0f,
62 0.0f, -1.0f, 0.0f
63 };
64 float[] sharpMatrix = new float[9];
65 for (int i = 0; i < 9; i++) {
66 sharpMatrix[i] = normalMatrix[i] + level * edgeMatrix[i];
67 }
68 return sharpMatrix;
69 }

162 public static void writeJpeg(Image img, String destFile, float quality, boolean sharpen, float sharpenLevel)
163 throws IOException {
164 sharpenMatrix = calculeSharpenMatrix(sharpenLevel);
165 BufferedImage imageRGBSharpen = null;
166 IIOImage iioImage = null;
167 BufferedImage image = SwingFXUtils.fromFXImage(img, null); // Get buffered image.
168
169 BufferedImage imageRGB = new BufferedImage(image.getWidth(), image.getHeight(), BufferedImage.OPAQUE); // Remove alpha-channel from buffered image.
170
171 Graphics2D graphics = imageRGB.createGraphics();
172 graphics.drawImage(image, 0, 0, null);
173 if (sharpen) {
174 imageRGBSharpen = new BufferedImage(image.getWidth(), image.getHeight(), BufferedImage.TYPE_INT_RGB);
175 Kernel kernel = new Kernel(3, 3, sharpenMatrix);
176 ConvolveOp cop = new ConvolveOp(kernel,
177 ConvolveOp.EDGE_NO_OP,
178 null);
179 cop.filter(imageRGB, imageRGBSharpen);
180 }
181 ImageWriter writer = null;
182 FileImageOutputStream output = null;
183 try {
184 writer = ImageIO.getImageWritersByFormatName("jpeg").next();
185 ImageWriteParam param = writer.getDefaultWriteParam();
186 param.setCompressionMode(ImageWriteParam.MODE_EXPLICIT);
187 param.setCompressionQuality(quality);
188 output = new FileImageOutputStream(new File(destFile));
189 writer.setOutput(output);
190 if (sharpen) {
191 iioImage = new IIOImage(imageRGBSharpen, null, null);
192 } else {
193 iioImage = new IIOImage(imageRGB, null, null);
194 }
195 writer.write(null, iioImage, param);
196 } catch (IOException ex) {
197 throw ex;
198 } finally {
199 if (writer != null) {
200 writer.dispose();
201 }
202 if (output != null) {
203 output.close();
204 }
205 }
206 graphics.dispose();
207 }

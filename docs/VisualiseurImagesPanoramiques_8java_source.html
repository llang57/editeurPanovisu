104 public static Image imgTransformationImage(Image imgRect, int iRapport) {
105 int iLargeur = (int) imgRect.getWidth() / iRapport;
106 int iHauteur = iLargeur / 2 / iRapport;
107 WritableImage imgMercator = new WritableImage(iLargeur, iHauteur);
108 PixelReader prRect = imgRect.getPixelReader();
109 PixelWriter pwMercator = imgMercator.getPixelWriter();
110 for (int i = 0; i < iLargeur; i++) {
111 for (int j = 0; j < iHauteur; j++) {
112 //double phi = 2 * Math.atan(Math.exp(2.d * Math.PI / largeur * (largeur / 4.d - (double) j))) - Math.PI / 2.d;
113 double phi = Math.asin(2.d * (iHauteur / 2.d - j) / iHauteur);
114 int y2 = (int) (iHauteur * iRapport * (0.5d - phi / Math.PI));
115 if (y2 >= iHauteur * iRapport) {
116 y2 = iHauteur * iRapport - 1;
117 }
118 Color clPixel = prRect.getColor(i * iRapport, y2 * iRapport);
119 pwMercator.setColor(i, j, clPixel);
120 }
121 }
122 return imgMercator;
123 }

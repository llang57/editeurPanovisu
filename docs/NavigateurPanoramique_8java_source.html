146 public static Image imgTransformationImage(Image imgRect, int iRapport) {
147 int iLargeur = (int) imgRect.getWidth() / iRapport;
148 int iHauteur = iLargeur / 2 / iRapport;
149 WritableImage imgMercator = new WritableImage(iLargeur, iHauteur);
150 PixelReader prRect = imgRect.getPixelReader();
151 PixelWriter pwMercator = imgMercator.getPixelWriter();
152 for (int i = 0; i < iLargeur; i++) {
153 for (int j = 0; j < iHauteur; j++) {
154 //double phi = 2 * Math.atan(Math.exp(2.d * Math.PI / largeur * (largeur / 4.d - (double) j))) - Math.PI / 2.d;
155 double phi = Math.asin(2.d * (iHauteur / 2.d - j) / iHauteur);
156 int y2 = (int) (iHauteur * iRapport * (0.5d - phi / Math.PI));
157 if (y2 >= iHauteur * iRapport) {
158 y2 = iHauteur * iRapport - 1;
159 }
160 Color clPixel = prRect.getColor(i * iRapport, y2 * iRapport);
161 pwMercator.setColor(i, j, clPixel);
162 }
163 }
164 return imgMercator;
165 }

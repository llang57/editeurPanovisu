65 public String generateText(String prompt, int maxLength, double temperature) throws Exception {
66 if (apiKey == null || apiKey.isEmpty()) {
67 throw new IllegalStateException("Clé API Hugging Face non configurée");
68 }
69
70 String endpoint = API_BASE_URL + modelName;
71 URI uri = URI.create(endpoint);
72 HttpURLConnection conn = (HttpURLConnection) uri.toURL().openConnection();
73
74 try {
75 // Configuration de la requête
76 conn.setRequestMethod("POST");
77 conn.setRequestProperty("Authorization", "Bearer " + apiKey);
78 conn.setRequestProperty("Content-Type", "application/json");
79 conn.setDoOutput(true);
80
81 // Corps de la requête JSON
82 String jsonInput = String.format(
83 "{\"inputs\": \"%s\", \"parameters\": {\"max_length\": %d, \"temperature\": %.2f}}",
84 escapeJson(prompt), maxLength, temperature
85 );
86
87 // Envoi de la requête
88 try (OutputStream os = conn.getOutputStream()) {
89 byte[] input = jsonInput.getBytes(StandardCharsets.UTF_8);
90 os.write(input, 0, input.length);
91 }
92
93 // Lecture de la réponse
94 int responseCode = conn.getResponseCode();
95
96 if (responseCode == HttpURLConnection.HTTP_OK) {
97 try (BufferedReader br = new BufferedReader(
98 new InputStreamReader(conn.getInputStream(), StandardCharsets.UTF_8))) {
99 StringBuilder response = new StringBuilder();
100 String line;
101 while ((line = br.readLine()) != null) {
102 response.append(line);
103 }
104
105 // Parsing simple de la réponse JSON
106 // Format: [{"generated_text": "..."}]
107 String jsonResponse = response.toString();
108 return extractGeneratedText(jsonResponse);
109 }
110 } else {
111 try (BufferedReader br = new BufferedReader(
112 new InputStreamReader(conn.getErrorStream(), StandardCharsets.UTF_8))) {
113 StringBuilder error = new StringBuilder();
114 String line;
115 while ((line = br.readLine()) != null) {
116 error.append(line);
117 }
118 throw new Exception("Erreur API Hugging Face (" + responseCode + "): " + error);
119 }
120 }
121 } finally {
122 conn.disconnect();
123 }

217 private String extractGeneratedText(String jsonResponse) {
218 // Parsing simple - pour un parsing robuste, utiliser une lib JSON
219 String marker = "\"generated_text\":";
220 int startIdx = jsonResponse.indexOf(marker);
221 if (startIdx == -1) {
222 return jsonResponse; // Retourner la réponse brute si format non reconnu
223 }
224
225 startIdx += marker.length();
226 // Trouver le début de la chaîne
227 while (startIdx < jsonResponse.length() && jsonResponse.charAt(startIdx) != '"') {
228 startIdx++;
229 }
230 startIdx++; // Passer le guillemet
231
232 // Trouver la fin de la chaîne
233 int endIdx = startIdx;
234 while (endIdx < jsonResponse.length()) {
235 if (jsonResponse.charAt(endIdx) == '"' && jsonResponse.charAt(endIdx - 1) != '\\') {
236 break;
237 }
238 endIdx++;
239 }
240
241 return jsonResponse.substring(startIdx, endIdx);
242 }

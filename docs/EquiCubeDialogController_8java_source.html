306 public Task tskTraitement() {
307 return new Task() {
308 @Override
309 protected Object call() throws Exception {
310 apTransformations.setCursor(Cursor.WAIT);
311 Platform.runLater(() -> {
312 for (int ij = 0; ij < lvListeFichier.getItems().size(); ij++) {
313 String strNomFich1 = (String) lvListeFichier.getItems().get(ij);
314 lvListeFichier.getItems().set(ij, "A Traiter => " + strNomFich1);
315 }
316 });
317 updateProgress(0.0001f, lvListeFichier.getItems().size());
318 for (int i1 = 0; i1 < lvListeFichier.getItems().size(); i1++) {
319 Thread.sleep(200);
320 updateMessage("Traitement en cours " + (i1 + 1) + "/" + lvListeFichier.getItems().size());
321 String strNomFich = ((String) lvListeFichier.getItems().get(i1)).split("> ")[1];
322 final int ii = i1;
323 Platform.runLater(() -> {
324 lvListeFichier.getItems().set(ii, "Traitement en cours => " + strNomFich);
325 });
326 traiteFichier(strNomFich, i1);
327 Thread.sleep(100);
328
329 updateProgress(i1 + 1, lvListeFichier.getItems().size());
330 Platform.runLater(() -> {
331 lvListeFichier.getItems().set(ii, "Traité => " + strNomFich);
332 });
333 Thread.sleep(100);
334 }
335 apTransformations.setCursor(Cursor.DEFAULT);
336 return true;
337 }

359 private File[] choixFichiers() {
360 File[] fileLstFich = null;
361 FileChooser fcRepertChoix = new FileChooser();
362 fcRepertChoix.getExtensionFilters().addAll(
363 new FileChooser.ExtensionFilter("fichier Image (*.jpg|*.bmp|*.tif)", "*.jpg", "*.bmp", "*.tif")
364 );
365 File fileRepert = new File(strRepertFichier + File.separator);
366 fcRepertChoix.setInitialDirectory(fileRepert);
367 List<File> fileListe = fcRepertChoix.showOpenMultipleDialog(stTransformations);
368 int i = 0;
369 boolean bAttention = false;
370 if (fileListe != null) {
371 File[] fileLstFich1 = new File[fileListe.size()];
372 for (File fileLst : fileListe) {
373 if (i == 0) {
374 strRepertFichier = fileLst.getParent();
375 }
376 String strNomFich = fileLst.getAbsolutePath();
377 String strExtension;
378 strExtension = strNomFich.substring(strNomFich.lastIndexOf(".") + 1, strNomFich.length()).toLowerCase();
379 Image img = null;
380 if (!strExtension.equals("tif")) {
381 img = new Image("file:" + strNomFich);
382 } else {
383 try {
384 img = ReadWriteImage.readTiff(strNomFich);
385 } catch (IOException ex) {
386 Logger.getLogger(EquiCubeDialogController.class.getName()).log(Level.SEVERE, null, ex);
387 }
388 }
389 if (strTypeTransformation.equals(EquiCubeDialogController.EQUI2CUBE)) {
390 if (img.getWidth() == 2 * img.getHeight()) {
391 fileLstFich1[i] = fileLst;
392 i++;
393 } else {
394 bAttention = true;
395 }
396 } else {
397 if (img.getWidth() == img.getHeight()) {
398 String strNom = fileLst.getAbsolutePath().substring(0, fileLst.getAbsolutePath().length() - 6);
399 boolean bTrouve = false;
400 for (int j = 0; j < i; j++) {
401 String strNom1 = fileLstFich1[j].getAbsolutePath().substring(0, fileLst.getAbsolutePath().length() - 6);
402 if (strNom.equals(strNom1)) {
403 bTrouve = true;
404 }
405 }
406 if (!bTrouve) {
407 fileLstFich1[i] = fileLst;
408 i++;
409 }
410 } else {
411 bAttention = true;
412 }
413
414 }
415 }
416 if (bAttention) {
417 Alert alert = new Alert(AlertType.ERROR);
418 alert.setTitle(rbLocalisation.getString("transformation.traiteImages"));
419 alert.setHeaderText(null);
420 alert.setContentText(rbLocalisation.getString("transformation.traiteImagesType"));
421 alert.showAndWait();
422 }
423 fileLstFich = new File[i];
424 System.arraycopy(fileLstFich1, 0, fileLstFich, 0, i);
425 lblDragDropE2C.setVisible(false);
426 }
427
428 return fileLstFich;
429 }

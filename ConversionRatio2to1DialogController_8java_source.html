699 private void previsualiserSelection() {
700 File fichierSelectionne = lvListeFichiers.getSelectionModel().getSelectedItem();
701
702 if (fichierSelectionne == null) {
703 if (olFichiers.isEmpty()) {
704 afficherErreur(rbLocalisation.getString("conversion.erreurAucunFichier"));
705 return;
706 }
707 fichierSelectionne = olFichiers.get(0);
708 }
709
710 // Récupérer les paramètres individuels existants pour ce fichier
711 String cheminFichier = fichierSelectionne.getAbsolutePath();
712 double[] parametresExistants = parametresIndividuels.get(cheminFichier);
713
714 String typeRemplissage = "noir";
715 double positionVerticale = 0.5;
716
717 if (parametresExistants != null) {
718 // Charger les paramètres sauvegardés
719 positionVerticale = parametresExistants[0];
720 int typeRemplissageIndex = (int) parametresExistants[1];
721
722 switch (typeRemplissageIndex) {
723 case 0: typeRemplissage = "noir"; break;
724 case 1: typeRemplissage = "flou"; break;
725 case 2: typeRemplissage = "miroir"; break;
726 case 3: typeRemplissage = "couleurMoyenne"; break;
727 }
728 } else {
729 // Utiliser les paramètres globaux actuels
730 if (rbRemplissageFlou.isSelected()) {
731 typeRemplissage = "flou";
732 } else if (rbRemplissageMiroir.isSelected()) {
733 typeRemplissage = "miroir";
734 } else if (rbRemplissageCouleurMoyenne.isSelected()) {
735 typeRemplissage = "couleurMoyenne";
736 }
737 }
738
739 // Ouvrir le dialogue de prévisualisation avec les paramètres chargés
740 PrevisualiserConversionDialog dialog = new PrevisualiserConversionDialog(
741 fichierSelectionne, typeRemplissage
742 );
743 dialog.setPositionVerticale(positionVerticale);
744
745 if (dialog.afficher()) {
746 // L'utilisateur a validé, sauvegarder les paramètres individuels
747 typeRemplissage = dialog.getTypeRemplissage();
748 positionVerticale = dialog.getPositionVerticale();
749
750 // Convertir le type de remplissage en index
751 int typeRemplissageIndex = 0;
752 switch (typeRemplissage.toLowerCase()) {
753 case "flou": typeRemplissageIndex = 1; break;
754 case "miroir": typeRemplissageIndex = 2; break;
755 case "couleurmoyenne": typeRemplissageIndex = 3; break;
756 default: typeRemplissageIndex = 0; // noir
757 }
758
759 // Sauvegarder dans la map
760 parametresIndividuels.put(cheminFichier, new double[] {positionVerticale, typeRemplissageIndex});
761 }
762 }

832 private boolean estImageValide(File file) {
833 if (!file.exists() || !file.isFile()) {
834 return false;
835 }
836

1145 private void appliquerRemplissageFlou(Graphics2D g2d, BufferedImage imageSource,
1146 int largeurFinale, int hauteurFinale, double positionVerticale) {

1148 BufferedImage imageFond = new BufferedImage(largeurFinale, hauteurFinale, BufferedImage.TYPE_INT_RGB);
1149 Graphics2D gFond = imageFond.createGraphics();
1150
1151 // Étirer l'image source sur tout le fond
1152 gFond.drawImage(imageSource, 0, 0, largeurFinale, hauteurFinale, null);
1153 gFond.dispose();
1154
1155 // Appliquer un flou gaussien TRÈS FORT avec un kernel 9×9
1156 float[] matrix9x9 = new float[81];
1157 int kernelSize = 9;
1158 float sigma = 5.0f; // Augmenter sigma pour un flou plus fort
1159 float sum = 0.0f;
1160 int center = kernelSize / 2;
1161
1162 // Générer un kernel gaussien 9×9
1163 for (int y = 0; y < kernelSize; y++) {
1164 for (int x = 0; x < kernelSize; x++) {
1165 int dx = x - center;
1166 int dy = y - center;
1167 float value = (float) Math.exp(-(dx*dx + dy*dy) / (2*sigma*sigma));
1168 matrix9x9[y * kernelSize + x] = value;
1169 sum += value;
1170 }
1171 }
1172
1173 // Normaliser le kernel
1174 for (int i = 0; i < matrix9x9.length; i++) {
1175 matrix9x9[i] /= sum;
1176 }
1177
1178 Kernel kernel = new Kernel(kernelSize, kernelSize, matrix9x9);
1179 ConvolveOp op = new ConvolveOp(kernel, ConvolveOp.EDGE_NO_OP, null);
1180
1181 // Appliquer le flou 15 fois pour un effet TRÈS prononcé
1182 BufferedImage temp = imageFond;
1183 for (int i = 0; i < 15; i++) {
1184 BufferedImage temp2 = new BufferedImage(largeurFinale, hauteurFinale, BufferedImage.TYPE_INT_RGB);
1185 temp2 = op.filter(temp, temp2);
1186 temp = temp2;
1187 }

1189 // Dessiner le fond flouté avec une légère transparence
1190 g2d.drawImage(temp, 0, 0, null);
1191
1192 // Dessiner l'image source par-dessus avec la position verticale personnalisée
1193 int x = (largeurFinale - imageSource.getWidth()) / 2;
1194 int y = calculerPositionY(hauteurFinale, imageSource.getHeight(), positionVerticale);
1195 g2d.drawImage(imageSource, x, y, null);

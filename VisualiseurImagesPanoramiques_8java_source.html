142 public static Image imgTransformationImage(Image imgRect, int iRapport) {
143 int iLargeur = (int) imgRect.getWidth() / iRapport;
144 int iHauteur = iLargeur / 2 / iRapport;
145
146 try {
147 // Tentative GPU Bicubic pour affichage visite (plus rapide)
148 Image resized = ImageResizeGPU.resizeAuto(
149 imgRect,
150 iLargeur,
151 iHauteur,
152 InterpolationMethod.BICUBIC
153 );
154 Logger.getLogger(VisualiseurImagesPanoramiques.class.getName()).log(Level.FINE,
155 String.format("imgTransformationImage GPU: %dx%d → %dx%d (Bicubic)",
156 (int) imgRect.getWidth(), (int) imgRect.getHeight(),
157 iLargeur, iHauteur)
158 );
159 return resized;
160
161 } catch (Exception e) {
162 // Fallback CPU : boucle pixel par pixel (transformation Mercator)
163 Logger.getLogger(VisualiseurImagesPanoramiques.class.getName()).log(Level.WARNING,
164 "Erreur GPU imgTransformationImage, fallback CPU: " + e.getMessage()
165 );
166 return imgTransformationImageCPU(imgRect, iRapport, iLargeur, iHauteur);
167 }
168 }

173 private static Image imgTransformationImageCPU(Image imgRect, int iRapport, int iLargeur, int iHauteur) {
174 WritableImage imgMercator = new WritableImage(iLargeur, iHauteur);
175 PixelReader prRect = imgRect.getPixelReader();
176 PixelWriter pwMercator = imgMercator.getPixelWriter();
177 for (int i = 0; i < iLargeur; i++) {
178 for (int j = 0; j < iHauteur; j++) {
179 //double phi = 2 * Math.atan(Math.exp(2.d * Math.PI / largeur * (largeur / 4.d - (double) j))) - Math.PI / 2.d;
180 double phi = Math.asin(2.d * (iHauteur / 2.d - j) / iHauteur);
181 int y2 = (int) (iHauteur * iRapport * (0.5d - phi / Math.PI));
182 if (y2 >= iHauteur * iRapport) {
183 y2 = iHauteur * iRapport - 1;
184 }
185 Color clPixel = prRect.getColor(i * iRapport, y2 * iRapport);
186 pwMercator.setColor(i, j, clPixel);
187 }
188 }
189 return imgMercator;
190 }

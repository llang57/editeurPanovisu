55 private static float[] calculeSharpenMatrix(float level) {
56 float[] normalMatrix = {
57 0.0f, 0.0f, 0.0f,
58 0.0f, 1.0f, 0.0f,
59 0.0f, 0.0f, 0.0f
60 };
61 float[] edgeMatrix = {
62 0.0f, -1.0f, 0.0f,
63 -1.0f, 4.0f, -1.0f,
64 0.0f, -1.0f, 0.0f
65 };
66 float[] sharpMatrix = new float[9];
67 for (int i = 0; i < 9; i++) {
68 sharpMatrix[i] = normalMatrix[i] + level * edgeMatrix[i];
69 }
70 return sharpMatrix;
71 }

107 throws ImagingException, IOException {
108 File file = new File(strNomFich);
109 BufferedImage imageRGBSharpen = null;
110 BufferedImage imageRGB = SwingFXUtils.fromFXImage(imgImage, null);
111
112 Graphics2D graphics = imageRGB.createGraphics();
113 graphics.drawImage(imageRGB, 0, 0, null);
114 if (bSharpen) {
115 imageRGBSharpen = new BufferedImage(imageRGB.getWidth(), imageRGB.getHeight(), BufferedImage.TYPE_INT_RGB);
116 Kernel kernel = new Kernel(3, 3, sharpenMatrix);

154 public static Image resizeImage(Image img, int newW, int newH) {
155 try {
156 // Essayer GPU d'abord (Bicubic)
157 Image resized = ImageResizeGPU.resizeAuto(
158 img,
159 newW,
160 newH,
161 InterpolationMethod.BICUBIC
162 );
163
164 Logger.getLogger(ReadWriteImage.class.getName()).log(
165 Level.FINE,
166 String.format("Resize GPU: %dx%d → %dx%d (Bicubic)",
167 (int)img.getWidth(), (int)img.getHeight(), newW, newH
168 )
169 );
170
171 return resized;
172
173 } catch (Exception e) {
174 // Fallback sur Thumbnails (CPU) en cas d'erreur GPU
175 Logger.getLogger(ReadWriteImage.class.getName()).log(
176 Level.WARNING,
177 "Erreur GPU resize, fallback Thumbnails: " + e.getMessage()
178 );
179
180 BufferedImage image = SwingFXUtils.fromFXImage(img, null);
181 try {
182 BufferedImage imgRetour = Thumbnails.of(image).size(newW, newH).asBufferedImage();
183 return SwingFXUtils.toFXImage(imgRetour, null);
184 } catch (IOException ex) {
185 Logger.getLogger(ReadWriteImage.class.getName()).log(Level.SEVERE, null, ex);
186 }
187 return null;
188 }

200 public static void writeJpeg(Image img, String destFile, float quality, boolean sharpen, float sharpenLevel)
201 throws IOException {
202 sharpenMatrix = calculeSharpenMatrix(sharpenLevel);
203 BufferedImage imageRGBSharpen = null;
204 IIOImage iioImage = null;
205 BufferedImage image = SwingFXUtils.fromFXImage(img, null); // Get buffered image.
206
207 // Vérification que l'image a été correctement chargée
208 if (image == null || image.getWidth() == 0 || image.getHeight() == 0) {
209 System.err.println("⚠️ writeJpeg: image invalide ou dimensions nulles pour " + destFile);
210 return; // On ignore cette image plutôt que de crasher
211 }
212
213 BufferedImage imageRGB = new BufferedImage(image.getWidth(), image.getHeight(), BufferedImage.OPAQUE); // Remove alpha-channel from buffered image.
214
215 Graphics2D graphics = imageRGB.createGraphics();
216 graphics.drawImage(image, 0, 0, null);
217 if (sharpen) {
218 imageRGBSharpen = new BufferedImage(image.getWidth(), image.getHeight(), BufferedImage.TYPE_INT_RGB);
219 Kernel kernel = new Kernel(3, 3, sharpenMatrix);
220 ConvolveOp cop = new ConvolveOp(kernel,
221 ConvolveOp.EDGE_NO_OP,
222 null);
223 cop.filter(imageRGB, imageRGBSharpen);
224 }
225 ImageWriter writer = null;
226 FileImageOutputStream output = null;
227 try {
228 writer = ImageIO.getImageWritersByFormatName("jpeg").next();
229 ImageWriteParam param = writer.getDefaultWriteParam();
230 param.setCompressionMode(ImageWriteParam.MODE_EXPLICIT);
231 param.setCompressionQuality(quality);
232 output = new FileImageOutputStream(new File(destFile));
233 writer.setOutput(output);
234 if (sharpen) {
235 iioImage = new IIOImage(imageRGBSharpen, null, null);
236 } else {
237 iioImage = new IIOImage(imageRGB, null, null);
238 }
239 writer.write(null, iioImage, param);
240 } catch (IOException ex) {
241 throw ex;
242 } finally {
243 if (writer != null) {
244 writer.dispose();
245 }
246 if (output != null) {
247 output.close();
248 }
249 }
250 graphics.dispose();
251 }

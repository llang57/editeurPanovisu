316 public Task<Boolean> tskTraitement() {
317 return new Task<Boolean>() {
318 @Override
319 protected Boolean call() throws Exception {
320 apTransformations.setCursor(Cursor.WAIT);
321 Platform.runLater(() -> {
322 for (int ij = 0; ij < lvListeFichier.getItems().size(); ij++) {
323 String strNomFich1 = (String) lvListeFichier.getItems().get(ij);
324 lvListeFichier.getItems().set(ij, "A Traiter => " + strNomFich1);
325 }
326 });
327 updateProgress(0.0001f, lvListeFichier.getItems().size());
328 for (int i1 = 0; i1 < lvListeFichier.getItems().size(); i1++) {
329 Thread.sleep(200);
330 updateMessage("Traitement en cours " + (i1 + 1) + "/" + lvListeFichier.getItems().size());
331 String strNomFich = ((String) lvListeFichier.getItems().get(i1)).split("> ")[1];
332 final int ii = i1;
333 Platform.runLater(() -> {
334 lvListeFichier.getItems().set(ii, "Traitement en cours => " + strNomFich);
335 });
336 traiteFichier(strNomFich, i1);
337 Thread.sleep(100);
338
339 updateProgress(i1 + 1, lvListeFichier.getItems().size());
340 Platform.runLater(() -> {
341 lvListeFichier.getItems().set(ii, "Traité => " + strNomFich);
342 });
343 Thread.sleep(100);
344 }
345 apTransformations.setCursor(Cursor.DEFAULT);
346 return true;
347 }

369 private File[] choixFichiers() {
370 File[] fileLstFich = null;
371 FileChooser fcRepertChoix = new FileChooser();
372 fcRepertChoix.getExtensionFilters().addAll(
373 new FileChooser.ExtensionFilter("fichier Image (*.jpg|*.bmp|*.tif)", "*.jpg", "*.bmp", "*.tif")
374 );
375 File fileRepert = new File(strRepertFichier + File.separator);
376 fcRepertChoix.setInitialDirectory(fileRepert);
377 List<File> fileListe = fcRepertChoix.showOpenMultipleDialog(stTransformations);
378 int i = 0;
379 boolean bAttention = false;
380 if (fileListe != null) {
381 File[] fileLstFich1 = new File[fileListe.size()];
382 for (File fileLst : fileListe) {
383 if (i == 0) {
384 strRepertFichier = fileLst.getParent();
385 }
386 String strNomFich = fileLst.getAbsolutePath();
387 String strExtension;
388 strExtension = strNomFich.substring(strNomFich.lastIndexOf(".") + 1, strNomFich.length()).toLowerCase();
389 Image img = null;
390 if (!strExtension.equals("tif")) {
391 img = new Image("file:" + strNomFich);
392 } else {
393 try {
394 img = ReadWriteImage.readTiff(strNomFich);
395 } catch (IOException ex) {
396 Logger.getLogger(EquiCubeDialogController.class.getName()).log(Level.SEVERE, null, ex);
397 }
398 }
399 if (strTypeTransformation.equals(EquiCubeDialogController.EQUI2CUBE)) {
400 if (img.getWidth() == 2 * img.getHeight()) {
401 fileLstFich1[i] = fileLst;
402 i++;
403 } else {
404 bAttention = true;
405 }
406 } else {
407 if (img.getWidth() == img.getHeight()) {
408 String strNom = fileLst.getAbsolutePath().substring(0, fileLst.getAbsolutePath().length() - 6);
409 boolean bTrouve = false;
410 for (int j = 0; j < i; j++) {
411 String strNom1 = fileLstFich1[j].getAbsolutePath().substring(0, fileLst.getAbsolutePath().length() - 6);
412 if (strNom.equals(strNom1)) {
413 bTrouve = true;
414 }
415 }
416 if (!bTrouve) {
417 fileLstFich1[i] = fileLst;
418 i++;
419 }
420 } else {
421 bAttention = true;
422 }
423
424 }
425 }
426 if (bAttention) {
427 Alert alert = new Alert(AlertType.ERROR);
428 alert.setTitle(rbLocalisation.getString("transformation.traiteImages"));
429 alert.setHeaderText(null);
430 alert.setContentText(rbLocalisation.getString("transformation.traiteImagesType"));
431 alert.showAndWait();
432 }
433 fileLstFich = new File[i];
434 System.arraycopy(fileLstFich1, 0, fileLstFich, 0, i);
435 lblDragDropE2C.setVisible(false);
436 }
437
438 return fileLstFich;
439 }

80 public boolean afficher() {
81 try {
82 // Charger l'image source complète
83 BufferedImage imageComplete = ImageIO.read(fichierSource);
84 if (imageComplete == null) {
85 return false;
86 }
87
88 // Créer une vignette pour la prévisualisation (max 800px de largeur)
89 imageSource = creerVignette(imageComplete, 800);
90
91 // Libérer l'image complète de la mémoire
92 imageComplete.flush();
93 imageComplete = null;
94
95 // Créer et configurer le stage
96 stage = new Stage();
97 stage.initModality(Modality.APPLICATION_MODAL);
98 stage.initOwner(EditeurPanovisu.getStPrincipal());
99 stage.setTitle(rbLocalisation.getString("conversion.previsualisation") + " - " + fichierSource.getName());
100 stage.setResizable(true);
101
102 // Créer l'interface
103 creerInterface();
104
105 // Générer la prévisualisation initiale
106 genererPrevisualisation();
107
108 // Afficher et attendre
109 stage.showAndWait();

315 private BufferedImage creerVignette(BufferedImage imageOriginale, int largeurMax) {
316 int largeurSource = imageOriginale.getWidth();
317 int hauteurSource = imageOriginale.getHeight();
318
319 // Si l'image est déjà petite, la retourner telle quelle
320 if (largeurSource <= largeurMax) {
321 return imageOriginale;
322 }
323
324 // Calculer les nouvelles dimensions en gardant le ratio
325 int nouvelleLargeur = largeurMax;
326 int nouvelleHauteur = (int) ((double) hauteurSource * largeurMax / largeurSource);
327
328 // Créer la vignette avec interpolation bicubique
329 BufferedImage vignette = new BufferedImage(
330 nouvelleLargeur,
331 nouvelleHauteur,
332 BufferedImage.TYPE_INT_RGB
333 );
334
335 Graphics2D g2d = vignette.createGraphics();
336 g2d.setRenderingHint(RenderingHints.KEY_INTERPOLATION,
337 RenderingHints.VALUE_INTERPOLATION_BICUBIC);
338 g2d.setRenderingHint(RenderingHints.KEY_RENDERING,
339 RenderingHints.VALUE_RENDER_QUALITY);
340 g2d.setRenderingHint(RenderingHints.KEY_ANTIALIASING,
341 RenderingHints.VALUE_ANTIALIAS_ON);
342 g2d.drawImage(imageOriginale, 0, 0, nouvelleLargeur, nouvelleHauteur, null);
343 g2d.dispose();
344
345 return vignette;
346 }

351 private void genererPrevisualisation() {
352 piChargement.setVisible(true);
353 imageViewPreview.setVisible(false);
354
355 // Annuler le thread précédent s'il existe
356 if (threadPrevisualisation != null && threadPrevisualisation.isAlive()) {
357 threadPrevisualisation.interrupt();
358 }
359
360 // Exécuter dans un thread séparé pour ne pas bloquer l'UI
361 threadPrevisualisation = new Thread(() -> {
362 BufferedImage imageConvertie = null;
363 try {
364 imageConvertie = convertirAvecOptions(imageSource);
365
366 // Vérifier si le thread a été interrompu
367 if (Thread.currentThread().isInterrupted()) {
368 return;
369 }
370
371 // Convertir en JavaFX Image
372 Image fxImage = SwingFXUtils.toFXImage(imageConvertie, null);
373
374 // Libérer la mémoire
375 imageConvertie.flush();
376
377 // Mettre à jour l'UI dans le thread JavaFX
378 Platform.runLater(() -> {
379 imageViewPreview.setImage(fxImage);
380 vbContenu.getChildren().clear();
381 vbContenu.getChildren().add(imageViewPreview);
382 imageViewPreview.setVisible(true);
383 piChargement.setVisible(false);
384 });
385
386 } catch (Exception ex) {
387 if (!Thread.currentThread().isInterrupted()) {
388 Platform.runLater(() -> {
389 piChargement.setVisible(false);
390 Label lblErreur = new Label("Erreur : " + ex.getMessage());
391 lblErreur.setStyle("-fx-text-fill: red;");
392 vbContenu.getChildren().add(lblErreur);
393 });
394 }
395 } finally {
396 // Libérer la mémoire
397 if (imageConvertie != null) {
398 imageConvertie.flush();
399 imageConvertie = null;
400 }
401 System.gc();
402 }
403 });
404 threadPrevisualisation.setDaemon(true);
405 threadPrevisualisation.start();
406 }

411 private BufferedImage convertirAvecOptions(BufferedImage imageSource) {
412 int largeurSource = imageSource.getWidth();
413 int hauteurSource = imageSource.getHeight();
414 double ratioSource = (double) largeurSource / hauteurSource;
415
416 final double RATIO_CIBLE = 2.0;
417
418 int largeurFinale, hauteurFinale;
419
420 if (Math.abs(ratioSource - RATIO_CIBLE) < 0.01) {
421 return imageSource;
422 }
423
424 // Calculer les dimensions finales
425 if (ratioSource < RATIO_CIBLE) {
426 hauteurFinale = hauteurSource;
427 largeurFinale = (int) Math.round(hauteurFinale * RATIO_CIBLE);
428 } else {
429 largeurFinale = largeurSource;
430 hauteurFinale = (int) Math.round(largeurFinale / RATIO_CIBLE);
431 }
432
433 // Créer l'image finale
434 BufferedImage imageFinale = new BufferedImage(largeurFinale, hauteurFinale, BufferedImage.TYPE_INT_RGB);
435 Graphics2D g2d = imageFinale.createGraphics();
436
437 g2d.setRenderingHint(RenderingHints.KEY_INTERPOLATION, RenderingHints.VALUE_INTERPOLATION_BICUBIC);
438 g2d.setRenderingHint(RenderingHints.KEY_RENDERING, RenderingHints.VALUE_RENDER_QUALITY);
439 g2d.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_ON);
440
441 // Appliquer le remplissage selon le type
442 switch (typeRemplissage.toLowerCase()) {
443 case "noir":
444 appliquerRemplissageNoir(g2d, imageSource, largeurFinale, hauteurFinale);
445 break;
446 case "flou":
447 appliquerRemplissageFlou(g2d, imageSource, largeurFinale, hauteurFinale);
448 break;
449 case "miroir":
450 appliquerRemplissageMiroir(g2d, imageSource, largeurFinale, hauteurFinale);
451 break;
452 case "couleurmoyenne":
453 appliquerRemplissageCouleurMoyenne(g2d, imageSource, largeurFinale, hauteurFinale);
454 break;
455 }
456
457 g2d.dispose();
458
459 // Ajouter la grille de coordonnées
460 imageFinale = ajouterGrilleCoordonnees(imageFinale);
461
462 return imageFinale;
463 }

474 private BufferedImage ajouterGrilleCoordonnees(BufferedImage image) {
475 int largeur = image.getWidth();
476 int hauteur = image.getHeight();
477
478 // Créer une copie de l'image pour dessiner dessus
479 BufferedImage imageAvecGrille = new BufferedImage(largeur, hauteur, BufferedImage.TYPE_INT_RGB);
480 Graphics2D g2d = imageAvecGrille.createGraphics();
481
482 // Dessiner l'image de base
483 g2d.drawImage(image, 0, 0, null);
484
485 // Configuration pour la grille
486 g2d.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_ON);
487
488 // Couleur de la grille : gris semi-transparent (au lieu de blanc)
489 java.awt.Color couleurGrille = new java.awt.Color(160, 160, 160, 140);
490
491 // Dessiner les lignes verticales (longitudes)
492 // 360° de longitude répartis sur la largeur
493 // Traits pleins tous les 45°, pointillés tous les 15°
494 for (int longitude = -180; longitude <= 180; longitude += 15) {
495 float x = (float) ((longitude + 180) * largeur / 360.0);
496
497 g2d.setColor(couleurGrille);
498
499 if (longitude % 45 == 0) {
500 // Trait plein tous les 45° (au lieu de 20°)
501 g2d.setStroke(new java.awt.BasicStroke(1.5f));
502 g2d.drawLine((int) x, 0, (int) x, hauteur);
503 } else {
504 // Trait pointillé tous les 15° (au lieu de 10°)
505 float[] dashPattern = {5.0f, 5.0f};
506 g2d.setStroke(new java.awt.BasicStroke(1.0f,
507 java.awt.BasicStroke.CAP_BUTT,
508 java.awt.BasicStroke.JOIN_MITER,
509 10.0f, dashPattern, 0.0f));
510 g2d.drawLine((int) x, 0, (int) x, hauteur);
511 }
512 }
513
514 // Dessiner les lignes horizontales (latitudes)
515 // 180° de latitude répartis sur la hauteur (de +90° en haut à -90° en bas)
516 for (int latitude = -90; latitude <= 90; latitude += 10) {
517 float y = (float) ((90 - latitude) * hauteur / 180.0);
518
519 // L'équateur (latitude 0°) en jaune
520 if (latitude == 0) {
521 g2d.setColor(new java.awt.Color(255, 200, 0, 180)); // Jaune semi-transparent
522 g2d.setStroke(new java.awt.BasicStroke(2.0f)); // Plus épais
523 g2d.drawLine(0, (int) y, largeur, (int) y);
524 } else {
525 g2d.setColor(couleurGrille);
526
527 if (latitude % 20 == 0) {
528 // Trait plein tous les 20°
529 g2d.setStroke(new java.awt.BasicStroke(1.5f));
530 g2d.drawLine(0, (int) y, largeur, (int) y);
531 } else {
532 // Trait pointillé tous les 10°
533 float[] dashPattern = {5.0f, 5.0f};
534 g2d.setStroke(new java.awt.BasicStroke(1.0f,
535 java.awt.BasicStroke.CAP_BUTT,
536 java.awt.BasicStroke.JOIN_MITER,
537 10.0f, dashPattern, 0.0f));
538 g2d.drawLine(0, (int) y, largeur, (int) y);
539 }
540 }
541 }
542
543 g2d.dispose();
544 return imageAvecGrille;
545 }

571 private void appliquerRemplissageFlou(Graphics2D g2d, BufferedImage imageSource,
572 int largeurFinale, int hauteurFinale) {
573 BufferedImage imageFond = new BufferedImage(largeurFinale, hauteurFinale, BufferedImage.TYPE_INT_RGB);
574 Graphics2D gFond = imageFond.createGraphics();
575 gFond.drawImage(imageSource, 0, 0, largeurFinale, hauteurFinale, null);
576 gFond.dispose();
577
578 // Kernel gaussien 9×9
579 float[] matrix9x9 = new float[81];
580 int kernelSize = 9;
581 float sigma = 5.0f;
582 float sum = 0.0f;
583 int center = kernelSize / 2;
584
585 for (int y = 0; y < kernelSize; y++) {
586 for (int x = 0; x < kernelSize; x++) {
587 int dx = x - center;
588 int dy = y - center;
589 float value = (float) Math.exp(-(dx*dx + dy*dy) / (2*sigma*sigma));
590 matrix9x9[y * kernelSize + x] = value;
591 sum += value;
592 }
593 }
594
595 for (int i = 0; i < matrix9x9.length; i++) {
596 matrix9x9[i] /= sum;
597 }
598
599 Kernel kernel = new Kernel(kernelSize, kernelSize, matrix9x9);
600 ConvolveOp op = new ConvolveOp(kernel, ConvolveOp.EDGE_NO_OP, null);
601
602 BufferedImage temp = imageFond;
603 for (int i = 0; i < 10; i++) { // Réduire à 10 itérations pour la prévisualisation (plus rapide)
604 BufferedImage temp2 = new BufferedImage(largeurFinale, hauteurFinale, BufferedImage.TYPE_INT_RGB);
605 temp2 = op.filter(temp, temp2);
606 temp = temp2;
607 }
608
609 g2d.drawImage(temp, 0, 0, null);
610
611 int x = (largeurFinale - imageSource.getWidth()) / 2;
612 int y = calculerPositionY(hauteurFinale, imageSource.getHeight());
613 g2d.drawImage(imageSource, x, y, null);
614 }

619 private void appliquerRemplissageMiroir(Graphics2D g2d, BufferedImage imageSource,
620 int largeurFinale, int hauteurFinale) {
621 int largeurSource = imageSource.getWidth();
622 int hauteurSource = imageSource.getHeight();
623
624 int xCentre = (largeurFinale - largeurSource) / 2;
625 int yCentre = calculerPositionY(hauteurFinale, hauteurSource);
626
627 if (xCentre > 0) {
628 g2d.drawImage(imageSource, 0, yCentre, xCentre, yCentre + hauteurSource,
629 xCentre, 0, 0, hauteurSource, null);
630 g2d.drawImage(imageSource, xCentre + largeurSource, yCentre, largeurFinale, yCentre + hauteurSource,
631 largeurSource, 0, largeurSource - xCentre, hauteurSource, null);
632 }
633
634 if (yCentre > 0) {
635 g2d.drawImage(imageSource, xCentre, 0, xCentre + largeurSource, yCentre,
636 0, yCentre, largeurSource, 0, null);
637 }
638
639 int yBas = yCentre + hauteurSource;
640 if (yBas < hauteurFinale) {
641 g2d.drawImage(imageSource, xCentre, yBas, xCentre + largeurSource, hauteurFinale,
642 0, hauteurSource, largeurSource, hauteurSource - (hauteurFinale - yBas), null);
643 }
644
645 g2d.drawImage(imageSource, xCentre, yCentre, null);

651 private void appliquerRemplissageCouleurMoyenne(Graphics2D g2d, BufferedImage imageSource,
652 int largeurFinale, int hauteurFinale) {
653 int largeurSource = imageSource.getWidth();
654 int hauteurSource = imageSource.getHeight();
655
656 int xCentre = (largeurFinale - largeurSource) / 2;
657 int yCentre = calculerPositionY(hauteurFinale, hauteurSource);
658
659 int nbLignesAnalyse = Math.min(10, Math.min(largeurSource, hauteurSource) / 4);
660
661 if (yCentre > 0) {
662 java.awt.Color couleurHaut = calculerCouleurMoyenne(imageSource, 0, 0,
663 largeurSource, nbLignesAnalyse);
664 g2d.setColor(couleurHaut);
665 g2d.fillRect(0, 0, largeurFinale, yCentre);
666 }
667
668 int yBas = yCentre + hauteurSource;
669 if (yBas < hauteurFinale) {
670 java.awt.Color couleurBas = calculerCouleurMoyenne(imageSource, 0,
671 hauteurSource - nbLignesAnalyse,
672 largeurSource, nbLignesAnalyse);
673 g2d.setColor(couleurBas);
674 g2d.fillRect(0, yBas, largeurFinale, hauteurFinale - yBas);
675 }
676
677 if (xCentre > 0) {
678 java.awt.Color couleurGauche = calculerCouleurMoyenne(imageSource, 0, 0,
679 nbLignesAnalyse, hauteurSource);
680 g2d.setColor(couleurGauche);
681 g2d.fillRect(0, yCentre, xCentre, hauteurSource);
682
683 java.awt.Color couleurDroite = calculerCouleurMoyenne(imageSource,
